#include <bits/stdc++.h>

using namespace std;

string s;
map<string, int> temp;

bool cmp(string s1, string s2)
{
    if (temp[s1] == temp[s2])
    {
        return s1 < s2; // 相同次数，比较字典序
    }
    return temp[s1] > temp[s2]; // 直接先比较出现次数
}

int main()
{
    while (cin >> s)
    {
        string str;
        for (int i = 0; i < s.size(); i++)
        {
            if (isalpha(s[i]) || s[i] == ' ')
            {
                str += tolower(s[i]);
            }
        }

        for (int i = 0; i < str.size(); i++)
        {
            if (s[i] == '\'' && s[i + 1] == 's')
            {
                str[i] = str[i+1] = ' ';
            }
        }

        istringstream iss(str);
        string ss;
        while (iss >> ss)
        {
            temp[ss]++;
        }
    }

    vector<string> res;
    for (auto it = temp.begin(); it != temp.end(); it++)
    {
        res.push_back(it->first);
    }

    sort(res.begin(), res.end(), cmp);
    for (auto it : res)
    {
        cout << it << endl;
    }
}
